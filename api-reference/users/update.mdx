<ParamField body="first_name" type="string">Updated first name.</ParamField>

  Updated role: `AUDITOR` | `BUSINESS_ADMIN` | `BUSINESS_BOOKKEEPER` | `BUSINESS_OWNER` | `BUSINESS_USER` | `GUEST_USER` | `IT_ADMIN`
</ParamField>
<ParamField body="department_id" type="string">Updated department UUID.</ParamField>
