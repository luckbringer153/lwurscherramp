---
title: "Get Custom Field"
description: "Fetch a specific custom accounting field by ID."
api: "GET https://api.ramp.com/developer/v1/accounting/fields/{field_id}"
---

## Path Parameters

<ParamField path="field_id" type="string" required>
  The ID of the custom accounting field.
</ParamField>

## Response

<ResponseField name="accounting_connection_id" type="string" required />
<ResponseField name="id" type="string" />
<ResponseField name="ramp_id" type="string" />
<ResponseField name="name" type="string" />
<ResponseField name="display_name" type="string" required />
<ResponseField name="input_type" type="string" required>
  One of: `BOOLEAN`, `DATE`, `FREE_FORM_TEXT`, `SINGLE_CHOICE`.
</ResponseField>
<ResponseField name="is_active" type="boolean" />
<ResponseField name="is_splittable" type="boolean" />
<ResponseField name="is_required_for" type="array">
  `BILL`, `BILL_PAYMENT`, `INVOICE`, `PURCHASE_ORDER`, `REIMBURSEMENT`, `TRANSACTION`, `VENDOR_CREDIT`.
</ResponseField>
<ResponseField name="provider_name" type="string" required />
<ResponseField name="created_at" type="string (date)" />
<ResponseField name="updated_at" type="string (date)" />
