---
title: "List transactions"
description: "Retrieve a paginated list of card transactions."
api: "GET https://api.ramp.com/developer/v1/transactions"
---

**Required scope:** `transactions:read`

<Note>
  If `state` is not set, all transactions except declined transactions will be returned. Setting multiple ordering parameters is unsupported.
</Note>

## Query Parameters

<ParamField query="state" type="string">
  Filter by transaction state: `ALL` | `CLEARED` | `COMPLETION` | `DECLINED` | `ERROR` | `PENDING` | `PENDING_INITIATION`
</ParamField>

<ParamField query="sync_status" type="string">
  Filter by sync status: `NOT_SYNC_READY` | `SYNC_READY` | `SYNCED`
</ParamField>

<ParamField query="requested_to_sync" type="boolean">
  Filter to transactions that have been requested to sync (used with [Sync Button](/guides/sync-button)).
</ParamField>

<ParamField query="spend_limit_id" type="string">
  Filter by fund (limit) UUID.
</ParamField>

<ParamField query="card_id" type="string">
  Filter by card UUID.
</ParamField>

<ParamField query="user_id" type="string">
  Filter by cardholder UUID.
</ParamField>

<ParamField query="entity_id" type="string">
  Filter by business entity UUID.
</ParamField>

<ParamField query="from_date" type="string">
  Filter transactions from this date (ISO 8601).
</ParamField>

<ParamField query="to_date" type="string">
  Filter transactions up to this date (ISO 8601).
</ParamField>

<ParamField query="order_by_date_asc" type="boolean">
  Order results by date ascending.</ParamField>

<ParamField query="order_by_date_desc" type="boolean">
  Order results by date descending.
</ParamField>

<ParamField query="start" type="string">
  Pagination cursor from `page.next`.
</ParamField>

<ParamField query="page_size" type="integer">
  Number of results per page (default: 100, max: 100).
</ParamField>
